def _make_single_time_condition(dt_col, op, ref_dt_val=None):
    """
    Generate sql condition script of date comparison. For example, when ``op = '<'``, the condition 
    is ``dt_col < ref_dt_val`` when ``ref_dt_val`` is not ``None``. If ``ref_dt_val is None``, then 
    return ``None``.
    
    :param str dt_col: name of one date column with format ``yyyy-MM-dd``
    :param str op: ``op`` must be valid comparison operator, e.g., ``>``
    :param str ref_dt_val: name of the other date column with format ``yyyy-MM-dd``, defaults to None
    
    :return: sql comparison script
    """
    
    if ref_dt_val is None:
        return None

    script_template = """
    {dt_col} {op} '{ref_dt_val}'
    """
    script = script_template.format(
        dt_col=dt_col,
        op=op,
        ref_dt_val=ref_dt_val
    )

    return script

def _combine_conditions(conditions, op='and', where=True):
    """
    Combine a list of ``conditions`` using ``op``. Add ``where`` at the beginning if ``where`` is ``True``. 
    For instance, it might return ``cond_0 and cond_1``
    
    :param list conditions: list of conditions, such as produced by ``_make_single_time_condition``
    :param str op: logical ops

    :return: combined comparison sql script
    """

    sep = ' {} '.format(op)
    valid_conds = ['({})'.format(item) for item in conditions if item is not None]
    if not valid_conds:
        return ''
    cmb_conds = sep.join(valid_conds)
    if where:
        res = 'where {}'.format(cmb_conds)
    else:
        res = cmb_conds

    return res

def _make_entity_id_map_table(
    r, schema, prefix, node_to_join_tb_name, entity_id_map_tb_name, verbose=True):
    """
    Create the table mapping raw entity id to int id. The table is guaranteed to be consistent 
        for entity id mapping in sense that entity id is indexed in an incremental fashion.
    """

    script_template = """    
        drop table if exists {schema}.{prefix}node_dedup;
        create table {schema}.{prefix}node_dedup as
        select
            node_typ
            ,node_id
        from {node_to_join_tb_name}
        group by node_typ, node_id
        ;


        create table {target_table} as
        select
            *
            ,row_number() over (partition by 1) node_id_int
        from {schema}.{prefix}node_dedup
        ;

        drop table if exists {target_table}_tmp;
        create table {target_table}_tmp as
        with t as
        (
            select
                node_typ
                ,node_id
            from
            (
                select
                    t1.*
                    ,t2.node_id as flag
                from {schema}.{prefix}node_dedup as t1
                left join {target_table} as t2 on t1.node_typ=t2.node_typ and t1.node_id=t2.node_id
            ) t
            where flag is null
        ), t_max_idx as 
        (
            select
                max(node_id_int) max_node_id_int
            from {target_table}
        ), t_indexed as 
        (
            select
                t.*
                ,max_node_id_int + row_number() over (partition by 1) node_id_int
            from t cross join t_max_idx
        )

        select * from {target_table}
        union all
        select * from t_indexed
        ;

        drop table if exists {target_table};
        create table {target_table} as
        select * from {target_table}_tmp;

        drop table if exists {target_table}_tmp;

    """
    script = script_template.format(
        schema=schema,
        prefix=prefix,
        node_to_join_tb_name=node_to_join_tb_name,
        target_table=entity_id_map_tb_name
    )

    sql_from_str(r, script, verbose=verbose)
    get_table_stats(r, entity_id_map_tb_name, verbose=verbose)

[docs]    def write_features_to_database(self, df_features, output_table=None, fillna_val=-999, batch_n=10):
        """
        A utility function that writes computed graph features to database (Vega). It processes as follows: 
        
        1. Fill ``NA`` values with ``fillna_val`` (default -999);
        2. Write to database;
        3. Map entity integer id back to its raw id.
        
        :param Pandas.DataFrame df_features: computed graph features DataFrame that must contain the integer 
            entity id column named ``node_id_int``
        :param str output_table: the name of the output table, defaults to ``None``, in which case the output 
            table is named ``'{schema}.{prefix}graph_features'``
        :param float fillna_val: fill ``NA`` with ``fillna_val``, defaults to ``-999``
        :param int batch_n: the number of batches the data would be split to when written to database, 
            defaults to ``10``
        """
        output_table = 'graph_features' if output_table is None else output_table
        feature_tb_name = '{schema}.{prefix}{output_table}'.format(
            schema=self.schema, prefix=self.prefix, output_table=output_table
        )
        tmp_target_table = '{schema}.{prefix}{output_table}_tmp'.format(
            schema=self.schema, prefix=self.prefix, output_table=output_table
        )
        
        df_features_filled = df_features.fillna(fillna_val)
        
        dtypes = []
        for col, raw_typ in df_features_filled.dtypes.to_frame().reset_index().values:
            typ = str(raw_typ)
            if col != 'node_id_int':
                if 'int' in typ:
                    dtype = (col, 'bigint')
                elif 'float' in typ:
                    dtype = (col, 'float')
                else:
                    raise TypeError('Type {} is not supported!'.format(typ))
            else:
                dtype = (col, 'bigint', False)
            dtypes.append(dtype)
                
        n_rows = df_features_filled.shape[0]
                
        write_to_database(self.r, df_features_filled, tmp_target_table, 
                          dtypes, batch_n=batch_n, verbose=self._verbose)
        
        # map id_int to raw id
        script_template = """
        drop table if exists {target_table};
        create table {target_table} as
        select
            t1.node_typ
            ,t1.node_id
            ,t2.*
        from {entity_id_map_table} as t1
        inner join {tmp_target_table} as t2 on t1.node_id_int=t2.node_id_int
        ;
        """
        script = script_template.format(
            target_table=feature_tb_name,
            entity_id_map_table=self.entity_id_map_tb_name,
            tmp_target_table=tmp_target_table
        )
        
        logger.info('Start mapping and writing features to table: {}'.format(feature_tb_name))
        sql_from_str(self.r, script, verbose=self._verbose)
        logger.info('Finished mapping and writing features to table: {}'.format(feature_tb_name))
        get_table_stats(self.r, feature_tb_name, verbose=self._verbose)

    """
    Class to build and retrieve graph data whose node is a single type of entity and whose relation is defined by 
    ``link_entities``. 
    
    For instance, suppose that we aim to build a graph where nodes correspond to users, and a 
    pair of users is connected if they share the same ip, or device. In this case, we need provide as input a 
    source table in which each row contains ``<user_id, ip, device_id, event_time>`` (``event_time`` indicates 
    the time when a user used the corresponding device or ip, for example). Then, we set ``node_entity`` 
    to be ``user_id``, ``link_entities`` to be ``['ip', 'device_id']``, instantiate the class, and call its methods 
    to obtain the desired graph. 
    """
    
    def __init__(self, source_table, node_entity, link_entities, ts_col,
                 schema, prefix, okta_username, entity_id_map_tb_name=None, hotkey_th=None, verbose=True,
                 cluster='vega'):
        """
        :param str source_table: name of souce table in database; it must contain columns specified in 
            ``node_entity``, ``link_entities``, ``ts_col``.
        :param str node_entity: name of entity considered as node
        :param list link_entities: list of entities. A pair of ``node_entity`` 
            is connected if they share at least one entity in ``link_entities``
        :param str ts_col: column name of timestamp when the link is effective; should have format 
            ``yyyy-MM-dd hh:mm:ss``
        :param str schema: schema name, for instance, ``sandbox_analytics_au``
        :param str prefix: prefix of the resulting table, for intance, ``test_``
        :param str okta_username: okta username
        :param str entity_id_map_tb_name: the name of the table that maps entity raw id to integer id; 
            defaults to ``None``; if left default, the table is named 
            ``'{schema}.{prefix}entity_id_map'``
        :param int hotkey_th: defaults to ``None``, no filtering hotkey if set to ``None``; otherwise, filter hotkey if 
            occurrence greater than ``hotkey_th``
        :param bool verbose: ``True`` to print details, defaults to ``True``
        :param str cluster: defaults to ``vega``
        """
        
        self.source_table = source_table
        self.node_entity = node_entity
        self.link_entities = link_entities
        self.ts_col = ts_col
        self.schema=schema
        self.prefix=prefix
        self.hotkey_th = hotkey_th
        self.__internal_hotkey_th = 500
        if hotkey_th is not None:
            # mitigate performance issue when joining entities
            self.__internal_hotkey_th = min(hotkey_th, 500) 
            
        self.edge_list_tb_name = '{schema}.{prefix}edge_list'.format(schema=schema, prefix=prefix)
        entity_id_map_tb_name = 'entity_id_map' if entity_id_map_tb_name is None else entity_id_map_tb_name
        self.entity_id_map_tb_name = '{schema}.{prefix}{entity_id_map_tb_name}'.format(
            schema=self.schema, prefix=self.prefix, entity_id_map_tb_name=entity_id_map_tb_name
        )
        self.node_to_join_tb_name = '{schema}.{prefix}node_to_join'.format(
            schema=self.schema, prefix=self.prefix
        )
        self.link_entity_cnt_tb_name = '{schema}.{prefix}link_entity_cnt'.format(
            schema=self.schema, prefix=self.prefix
        )
        
        self.r = RedshiftHook(cluster=cluster, okta_username=okta_username)
        self._verbose = verbose
        self.__SEP = '\n'
        
    def __assert_date_range(self, start_date, end_date):
        """
        Assert that ``start_date < end_date``
        """
        assert start_date is None or end_date is None or start_date < end_date, 'start_date should be prior to end_date, but got {} >= {}'.format(start_date, end_date)
    
    def __make_time_window_condition(
        self, 
        start_date=None, end_date=None, 
        hotkey_th=None, where=True):
        
        hotkey_th = self.hotkey_th if hotkey_th is None else hotkey_th
        
        conditions = [
            _make_single_time_condition('node_ts_min', '>=', start_date),
            _make_single_time_condition('node_ts_max', '<', end_date),
        ]
        
        if hotkey_th is not None:
            filter_hotkey_condition = 'cnt <= {}'.format(hotkey_th)
            conditions.append(filter_hotkey_condition)

        time_window_condition = _combine_conditions(conditions, op='and', where=where)

        return time_window_condition
        
    def _make_node_to_join_table(self):
        """
        Create the table containing nodes to join
        """

        master_script_template = """
            drop table if exists {target_table}_tmp;
            create table {target_table}_tmp as 
            with base as 
            (
                {union_script}
            )

            select
                node_typ
                ,node_id
                ,edge_typ
                ,edge_id
                ,ts
            from base
            where node_id <> '' and node_id is not null and edge_id <> '' and edge_id is not null
            group by 
                node_typ
                ,node_id
                ,edge_typ
                ,edge_id
                ,ts
            ;
        """
        
        raw_union_script = []
        
        union_script_template = """
            select
                '{node_entity}'::text as node_typ
                ,{node_entity} as node_id
                ,'{link_entity}'::text as edge_typ
                ,{link_entity} as edge_id
                ,{ts_col} as ts
            from {source_table}
        """
        for link_entity in self.link_entities:
            _script = union_script_template.format(
                node_entity=self.node_entity,
                link_entity=link_entity,
                ts_col=self.ts_col,
                source_table=self.source_table
            )
            raw_union_script.append(_script)
            raw_union_script.append('union all')

        union_script = self.__SEP.join(raw_union_script[:-1])

        master_script = master_script_template.format(
            schema=self.schema,
            prefix=self.prefix,
            target_table=self.node_to_join_tb_name,
            union_script=union_script
        )
        
        sql_from_str(self.r, master_script, verbose=self._verbose)
        get_table_stats(self.r, '{}_tmp'.format(self.node_to_join_tb_name), verbose=self._verbose)
        
        logger.info('Start counting key occurrence')
        # count how many node_entity has co-occurrence with link_node
        cnt_link_entity_script_template = """
            drop table if exists {target_table};
            create table {target_table} as
            select
                edge_typ
                ,edge_id
                ,count(*) cnt
            from
            (
                select
                    edge_typ
                    ,edge_id
                    ,node_typ
                    ,node_id
                from {source_table}_tmp
                group by 
                    edge_typ
                    ,edge_id
                    ,node_typ
                    ,node_id
            ) t
            group by edge_typ, edge_id
            ;
        """
        cnt_link_entity_script = cnt_link_entity_script_template.format(
            target_table=self.link_entity_cnt_tb_name,
            source_table=self.node_to_join_tb_name
        )
        sql_from_str(self.r, cnt_link_entity_script, verbose=self._verbose)
        get_table_stats(self.r, self.link_entity_cnt_tb_name, verbose=self._verbose)    
        
        count_script_template = """
            drop table if exists {target_table};
            create table {target_table} as
            select
                node_typ
                ,node_id
                ,t1.edge_typ
                ,t1.edge_id
                ,ts
                ,cnt
            from {target_table}_tmp t1 
            inner join {link_entity_cnt_table} t2 on t1.edge_typ=t2.edge_typ and t1.edge_id=t2.edge_id
            ;
        """
        count_script = count_script_template.format(
            target_table=self.node_to_join_tb_name,
            link_entity_cnt_table=self.link_entity_cnt_tb_name
        )
        sql_from_str(self.r, count_script, verbose=self._verbose)
        get_table_stats(self.r, self.node_to_join_tb_name, verbose=self._verbose)    
            
        logger.info('End counting key occurrence')
        
    def _make_edge_list_table(self):
        script_template = """
            drop table if exists {target_table};
            create table {target_table} as 
            select
                node_typ_u
                ,node_typ_v
                ,node_id_raw_u
                ,node_id_raw_v
                ,node_id_u
                ,node_id_v
                ,case when node_ts_u > node_ts_v then node_ts_v else node_ts_u end as node_ts_min
                ,case when node_ts_u < node_ts_v then node_ts_v else node_ts_u end as node_ts_max
                ,cnt
            from
            (
                select
                    l.node_typ node_typ_u
                    ,r.node_typ node_typ_v
                    ,l.node_id node_id_raw_u
                    ,r.node_id node_id_raw_v
                    ,lm.node_id_int node_id_u
                    ,rm.node_id_int node_id_v
                    ,l.ts node_ts_u
                    ,r.ts node_ts_v
                    ,l.cnt
                from (select * from {schema}.{prefix}node_to_join where cnt <= {hotkey_th}) as l 
                inner join (select * from {schema}.{prefix}node_to_join where cnt <= {hotkey_th}) as r on l.edge_typ=r.edge_typ and l.edge_id=r.edge_id and l.node_id < r.node_id
                inner join {entity_id_map_table} as lm on l.node_typ=lm.node_typ and l.node_id=lm.node_id
                inner join {entity_id_map_table} as rm on r.node_typ=rm.node_typ and r.node_id=rm.node_id
            ) t
            group by 
                node_typ_u
                ,node_typ_v
                ,node_id_raw_u
                ,node_id_raw_v
                ,node_id_u
                ,node_id_v
                ,node_ts_u
                ,node_ts_v
                ,cnt
            ;
        """
         
        script = script_template.format(
            schema=self.schema,
            prefix=self.prefix,
            target_table=self.edge_list_tb_name,
            entity_id_map_table=self.entity_id_map_tb_name,
            hotkey_th=self.__internal_hotkey_th
        )
        
        sql_from_str(self.r, script, verbose=self._verbose)
        get_table_stats(self.r, self.edge_list_tb_name, verbose=self._verbose)

[docs]    def get_graph_data(self, output_path=None, 
                       start_date=None, end_date=None, 
                       force=True, hotkey_th=None):
        """
        Filter graph data within time range specified by ``start_date`` 
        and ``end_date`` and get graph data as Pandas.DataFrame from database. 
        If ``start_date`` is ``None``, it indicates *unbounded preceding*. 
        If ``end_date`` is ``None``, it indicates *unbounded following*.
        
        Attention: need first call :py:func:`giraffe.graph_data_builder.UniGraphDataBuilder.build_graph_data`
        
        :param str output_path: the path to save the obtained graph data, defaults to ``None``; if left 
            default, the output path is set to be ``./data/edge_list.txt``
        :param str start_date: should conform to the same format as ``ts_col``, defaults to ``None``
        :param str end_date: should conform to the same format as ``ts_col``, defaults to ``None``
        :param bool force: set to ``True`` to enforce a replacement of file even ``output_path`` exists file. 
            defaults to ``True``
        :param int hotkey_th: defaults to ``None``, no filtering hotkey if set to ``None``; otherwise, filter hotkey if 
            occurrence greater than ``hotkey_th``
        """
        
        self.__assert_date_range(start_date, end_date)
        
        if not force and os.path.isfile(output_path) :
            logger.warn('Already exists {}. Set force to True to enforce replacement of file.'.format(output_path))
            df_edge_list = pd.read_csv(
                output_path, 
                sep='\t', 
                header=None, 
                names=['node_id_u', 'node_id_v']
            )
            return df_edge_list
            
        # set time window boundaries
        time_window_condition = self.__make_time_window_condition(
            start_date=start_date, end_date=end_date, hotkey_th=hotkey_th
        )
        
        logger.info('time window boundary condition: {}'.format(time_window_condition))
        logger.info('Filter graph between {} (inclusive) and {} (exclusive)'.format(
            'unbounded preceding' if start_date is None else start_date, 
            'unbounded following' if end_date is None else end_date
        ))

        script_template = """
        select
            node_id_u
            ,node_id_v
        from {source_table}
        {time_window_condition}
        group by
            node_id_u
            ,node_id_v
        """
        script = script_template.format(
            source_table=self.edge_list_tb_name,
            time_window_condition=time_window_condition
        )
        
        logger.info('Start fetching graph data')
        df_edge_list = sql_from_str(self.r, script, verbose=self._verbose)
        logger.info('Finished fetching graph data')
        
        # mkdir
        if output_path is None:
            output_path = os.path.join('.', 'data', 'edge_list.txt')
        output_dir = os.path.dirname(output_path)
        if not os.path.exists(output_dir):
            logger.info('{} does not exist. Create it now.'.format(output_dir))
            os.makedirs(output_dir, exist_ok=True)
        logger.info('Start writing to {}'.format(output_path))
        df_edge_list[['node_id_u', 'node_id_v']].to_csv(output_path, sep='\t', index=False, header=False)
        logger.info('Finished writing to {}'.format(output_path))
               
        return df_edge_list

    

[docs]    def get_graph_delta_data(
        self,
        ref_start_date=None, ref_end_date=None, 
        start_date=None, end_date=None, hotkey_th=None,
        force=True):
        """
        Get incremental part of graph data ``[start_date, end_date[`` with respect to 
        referece/base graph data ``[ref_start_date, ref_end_date[``
        as Pandas.DataFrame from database.
        
        Attention: need first call :py:func:`giraffe.graph_data_builder.UniGraphDataBuilder.build_graph_data`
        
        Only allow to use this function when graph data have overlap. In non-overlapping cases, 
        call :py:func:`giraffe.graph_data_builder.UniGraphDataBuilder.get_graph_data` instead.
        
        :param str ref_start_date: should conform to the same format as ``ts_col``, defaults to ``None``
        :param str ref_end_date: should conform to the same format as ``ts_col``, defaults to ``None``
        :param str start_date: should conform to the same format as ``ts_col``, defaults to ``None``
        :param str end_date: should conform to the same format as ``ts_col``, defaults to ``None``
        :param int hotkey_th: defaults to ``None``, no filtering hotkey if set to ``None``; otherwise, filter hotkey if 
            occurrence greater than ``hotkey_th``
        :param bool force: set to ``True`` to enforce a replacement of file even ``output_path`` exists
        
        :return: <node_id_u, node_id_v, is_new>, where ``is_new == 1`` indicates new edges to add and edges to remove
            otherwise
        :rtype: Pandas.DataFrame
        """
        
        self.__assert_date_range(start_date, end_date)
        self.__assert_date_range(ref_start_date, ref_end_date)
        
        assert not(
            (end_date is not None and ref_start_date is not None and end_date <= ref_start_date)
            or 
            (start_date is not None and ref_end_date is not None and start_date >= ref_end_date)
        ), \
        'Detect non-overlapping case! Will do nothing here. Call get_graph_data instead.'
        
        # set time window boundaries
        time_window_condition = self.__make_time_window_condition(
            start_date=start_date, end_date=end_date, hotkey_th=hotkey_th
        )
        time_window_condition_ref = self.__make_time_window_condition(

    def __get_graph_delta_data_v2(
        self,
        ref_start_date=None, ref_end_date=None, 
        start_date=None, end_date=None, hotkey_th=None,
        force=True):
        """
        Get incremental part of graph data [``start_date``, ``end_date``[ with respect to 
        referece/base graph data [``ref_start_date``, ``ref_end_date``[
        as Pandas.DataFrame from database after calling ``build_graph_data``.
        
        Only allow to use this function when graph data have overlap. In non-overlapping cases, 
        call ``get_graph_data`` instead.
        
        :param bool force: set to ``True`` to enforce a replacement of file even ``output_path`` exists
        
        :return: <edge_list_add, edge_list_rm>, edge_list_* could be None
        """
        
        self.__assert_date_range(start_date, end_date)
        self.__assert_date_range(ref_start_date, ref_end_date)
        
        assert not(
            (end_date is not None and ref_start_date is not None and end_date <= ref_start_date)
            or 
            (start_date is not None and ref_end_date is not None and start_date >= ref_end_date)
        ), \
        'Detect non-overlapping case! Will do nothing here. Call get_graph_data instead.'
                
        # set time window for incrementally adding edge list
        add_conds_left = _combine_conditions(
            [
                _make_single_time_condition('node_ts_min', '>=', start_date),
                _make_single_time_condition('node_ts_min', '<', ref_start_date),
                # _make_single_time_condition('node_ts_max', '>=', start_date), redundant
                _make_single_time_condition('node_ts_max', '<', end_date)
            ],
            op='and', where=False
        )
        add_conds_right = _combine_conditions(
            [
                _make_single_time_condition('node_ts_max', '>=', ref_end_date),
                _make_single_time_condition('node_ts_max', '<', end_date),
                _make_single_time_condition('node_ts_min', '>=', start_date),
                # _make_single_time_condition('node_ts_min', '<', end_date) redundant
            ],
            op='and', where=False
        )
        add_conds = _combine_conditions([add_conds_left, add_conds_right], op='or', where=True)
        
        # set time window for incrementally removing edge list
        rm_conds_left = _combine_conditions(
            [
                _make_single_time_condition('node_ts_min', '>=', ref_start_date),
                _make_single_time_condition('node_ts_min', '<', start_date),
                _make_single_time_condition('node_ts_max', '<', ref_end_date)
            ],
            op='and', where=False
        )
        rm_conds_right = _combine_conditions(
            [
                _make_single_time_condition('node_ts_max', '>=', end_date),
                _make_single_time_condition('node_ts_max', '<', ref_end_date),
                _make_single_time_condition('node_ts_min', '>=', ref_start_date),
            ],
            op='and', where=False
        )
        rm_conds = _combine_conditions([rm_conds_left, rm_conds_right], op='or', where=True)
        
        # edge to add
        script_template = """
        select
            node_id_u
            ,node_id_v
        from {source_table}
        {cond}
        group by
            node_id_u
            ,node_id_v
        """
        
        logger.info('Start fetching graph delta data to add')
        logger.info('time window boundary condition: {}'.format(add_conds))
        script = script_template.format(
            source_table=self.edge_list_tb_name,
            cond=add_conds
        )
        df_edge_list_add = sql_from_str(self.r, script, verbose=self._verbose)
        logger.info('Finished fetching graph delta data to add')
        
        # get edge to remove and pay attention to not over-remove
        time_window_condition = self.__make_time_window_condition(start_date=start_date, end_date=end_date)
        
        script_template = """
        with t as
        (
            select
                node_id_u
                ,node_id_v
            from {source_table}
            {time_window_condition}
            group by
                node_id_u
                ,node_id_v
        ), t_rm as
        (
            select
                node_id_u
                ,node_id_v
            from {source_table}
            {rm_conds}
            group by
                node_id_u
                ,node_id_v
        )
        
        select
            node_id_u
            ,node_id_v
        from
        (
            select
                t1.*
                ,t2.node_id_u flag
            from t_rm as t1 left join t as t2 on t1.node_id_u=t2.node_id_u and t1.node_id_v=t2.node_id_v
        ) a
        where flag is null
        """
        
        logger.info('Start fetching graph delta data to remove')
        logger.info('time window boundary condition: {}'.format(rm_conds))
        script = script_template.format(
            source_table=self.edge_list_tb_name,
            rm_conds=rm_conds,
            time_window_condition=time_window_condition
        )
        df_edge_list_rm = sql_from_str(self.r, script, verbose=self._verbose)
        logger.info('Finished fetching graph delta data to remove')
        
        return df_edge_list_add, df_edge_list_rm
