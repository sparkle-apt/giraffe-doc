class _IndexMapper:
    """
    Help to organize id mapping
    """
    def __init__(self, raw2int_map=None):
        """
        :param dict raw2int_map: raw id to int id
        """
        
        if raw2int_map is None:
            self.raw2int_map = {}
            self.max_int_id = -1
        else:
            self.raw2int_map = copy.deepcopy(raw2int_map)
            self.max_int_id = max(self.raw2int_map.values())
            
        self.int2raw_map = {v:k for k,v in self.raw2int_map.items()}
    
    def raw2int(self, raw_id):
        try:
            return self.raw2int_map[raw_id]
        except KeyError as error:
            logger.error(error)
            raise
    
    def int2raw(self, int_id):
        try:
            return self.int2raw_map[int_id]
        except KeyError as error:
            logger.error(error)
            raise
            
    def add_item(self, raw_id):
        if raw_id not in self.raw2int_map:
            self.max_int_id += 1
            self.raw2int_map[raw_id] = self.max_int_id
            self.int2raw_map[self.max_int_id] = raw_id
        else:
            logger.info('Ignore adding item since {} already exists'.format(raw_id))
            
    def remove_item(self, raw_id):
        if raw_id in self.raw2int_map:
            int_id = self.raw2int_map[raw_id]
            del self.raw2int_map[raw_id]
            del self.int2raw_map[int_id]
        else:
            logger.info('Ignore removing item since {} does not exist'.format(raw_id))

    """
    This is a class for building graph features. It supports the following features:
    
    * pagerank: PageRank
    * approx_betweenness: approximate betweenness based on path sampling
    * est_betweenness: estimate betweenness, perform better in practice than approx_betweenness (Sanders, Geisberger, Schultes: Better Approximation of Betweenness Centrality)
    * degree: degree centrality
    * approx_closeness: approximate closeness (Cohen et al., Computing Classic Closeness Centrality, at Scale)
    * harmon_closeness: harmonic closeness, suitable for unconnected graphs
    * katz: Katz centrality
    * clustering_coef: local clustering coefficient
    * louvain: community detection algorithm Louvain
    * kcore: K core algorithm
    """
    def __init__(self, features=None, config=None):
        """
        :param list features: only accepts values from 
        
            .. code-block:: python
                
                [
                    'pagerank',
                    'approx_betweenness',
                    'est_betweenness',
                    'degree',
                    'approx_closeness',
                    'harmon_closeness',
                    'katz',
                    'clustering_coef',
                    'louvain',
                    'kcore'
                ]
            
            defaults to ``None``; if ``features=None``, then compute all supported features
        :param dict config: defaults to ``None``; if config for some feature not provided, 
            will use default config value; otherwise, overwrite default one. The default value is 
            
            .. code-block:: python
            
                {
                    'pagerank': {
                        'kwargs': {'damp': 0.85, 'tol': 1e-9}
                    },
                    'approx_betweenness': {
                        'kwargs': {'epsilon': 0.01, 'delta': 0.1, 'universalConstant': 1.0}
                    },
                    'est_betweenness': {
                        'args': [500],
                        'kwargs': {'normalized': True, 'parallel': True}
                    },
                    'degree': {
                        'kwargs': {'normalized': False, 'outDeg': True, 'ignoreSelfLoops': True}
                    },
                    'approx_closeness': {
                        'args': [1000],
                        'kwargs': {'epsilon': 0.1, 'normalized': True, 'type': 2}
                    },
                    'harmon_closeness': {
                        'kwargs': {'normalized': True}
                    },
                    'katz': {
                        'kwargs': {'alpha': 5e-4, 'beta': 0.1, 'tol': 1e-8}
                    },
                    'clustering_coef': {
                        'kwargs': {'turbo': False} # 'normalized': True, 
                    },
                    'louvain': {
                        'kwargs': {
                            'refine': False,
                            'gamma': 0.1,
                            # 'par': 'balance',
                            'maxIter': 32,
                            'turbo': True,
                            'recurse': True,
                        }
                    },
                    'kcore': {}
                }
        """

        self.__edge_list_reader = None
        self.__index_mapper = None

            'pagerank': {
                'kwargs': {'damp': 0.85, 'tol': 1e-9}
            },
            'approx_betweenness': {
                'kwargs': {'epsilon': 0.01, 'delta': 0.1, 'universalConstant': 1.0}
            },
            'est_betweenness': {

                'kwargs': {'normalized': True, 'parallel': True}
            },
            'degree': {
                'kwargs': {'normalized': False, 'outDeg': True, 'ignoreSelfLoops': True}
            },
            'approx_closeness': {
                'args': [1000],
                'kwargs': {'epsilon': 0.1, 'normalized': True, 'type': 2}
            },
            'harmon_closeness': {
                'kwargs': {'normalized': True}
            },
            'katz': {
                'kwargs': {'alpha': 5e-4, 'beta': 0.1, 'tol': 1e-8}
            },
            'clustering_coef': {
                'kwargs': {'turbo': False} # 'normalized': True, 
            },
            'louvain': {
                'kwargs': {
                    'refine': False,
                    'gamma': 0.1,
                    # 'par': 'balance',
                    'maxIter': 32,

                }

        self.__assert_features(features)
        self.__assert_config(config)

        self.__config = {name:self.__default_config[name] for name in self.__features}
        # update config

        """
        Read graph data and convert to ``NetworKit.Graph`` object, and obtain index mapping
        
        :param str input_path: input path of EdgeList file
            The EdgeList file format is a simple format that stores each nodeâ€™s adjacency array

        """
        
        logger.info('Start building graph')
        self.__edge_list_reader = nk.graphio.EdgeListReader(
            separator='\t', firstNode=0, commentPrefix='#', continuous=False, directed=False
        )
        self.__g = self.__edge_list_reader.read(input_path)
        
        logger.info('Building mapping from original node id to internal id')
        
        # Get mapping from node ids in nxG to node ids in G
        # self.__node_id2int = dict(
        #     (int(raw_id), id_int) for raw_id, id_int in self.__edge_list_reader.getNodeMap().items()
        # )
        # self.__int2node_id = {v:k for k,v in self.__node_id2int.items()}
        raw2int_map = dict(
            (int(raw_id), id_int) for raw_id, id_int in self.__edge_list_reader.getNodeMap().items()
        )
        self.__index_mapper = _IndexMapper(raw2int_map)
        logger.info('Finished building graph')    
        logger.info(nk.overview(self.__g))

        

[docs]    def add_edges(self, edge_list):
        """
        Add edges to the graph
        
        :param list edge_list: list of edges, represented by tuples of node id
        """
        
        for u, v in edge_list:
            # add node if not exist
            if u not in self.__index_mapper.raw2int_map:
                self.__index_mapper.add_item(u)
                self.__g.addNode()
            if v not in self.__index_mapper.raw2int_map:
                self.__index_mapper.add_item(v)
                self.__g.addNode()
            
            # add edge
            self.__g.addEdge(
                self.__index_mapper.raw2int(u),
                self.__index_mapper.raw2int(v)
            )

[docs]    def update_graph(self, df_edge_list_delta):
        """
        Update graph by adding new edges and removing old edges specified in ``df_edge_list_delta``, 
        which can be returned by calling the method ``get_graph_delta_data`` of 
        :py:class:`giraffe.graph_data_builder.UniGraphDataBuilder` or 
        :py:class:`giraffe.graph_data_builder.BiGraphDataBuilder`
        
        :param DataFrame df_edge_list_delta:  contain ``<node_id_u, node_id_v, is_new>``, 
            represented by tuples of raw id of node, 
            and a tag ``is_new``, indicating whether to add (``is_new = 1``) or remove (``is_new = 0``)
        """
        
        # update edges
        edge_list_add = df_edge_list_delta.loc[df_edge_list_delta.is_new == 1, ['node_id_u', 'node_id_v']].values
        edge_list_rm = df_edge_list_delta.loc[df_edge_list_delta.is_new == 0, ['node_id_u', 'node_id_v']].values
        logger.info('Start adding #{} edges'.format(edge_list_add.shape[0]))
        self.add_edges(edge_list_add)
        logger.info('End adding edges')
        logger.info('Start removing #{} edges'.format(edge_list_rm.shape[0]))
        self.remove_edges(edge_list_rm)
        logger.info('End removing edges')
        
        # remove isolated nodes and remap node index
        logger.info('Start removing isolated nodes and remapping node indices')
        for nd in self.__g.iterNodes():
            if self.__g.isIsolated(nd):
                self.__g.removeNode(nd)

        prev_cur_map = nk.graphtools.getContinuousNodeIds(self.__g)
        prev_raw2int_map = self.__index_mapper.raw2int_map
        raw2int_map = {
            raw_id:prev_cur_map[prev_int_id] for raw_id, prev_int_id in prev_raw2int_map.items() \
                if prev_int_id in prev_cur_map
        }
        self.__index_mapper = _IndexMapper(raw2int_map)
        self.__g = nk.graphtools.getCompactedGraph(self.__g, prev_cur_map)
        logger.info('End removing isolated nodes and remapping node indices')

    
    def __get_args(self, name):
        if name in self.__config:
            return self.__config[name].get('args', [])
        else:
            raise KeyError('{} not supported'.format(name))
    
    def __get_kwargs(self, name):
        if name in self.__config:
            return self.__config[name].get('kwargs', {})
        else:
            raise KeyError('{} not supported'.format(name))
    
    def __build_feature(self, name):
        if name in self.__name2func:
            logger.info('Building feature {}'.format(name))
            func = self.__name2func[name]
            args = self.__get_args(name)
            kwargs = self.__get_kwargs(name)
            feature = func(self.__g, *args, **kwargs)
            feature.run()
            res = feature.scores()
            return res
        else:
            raise KeyError('{} not supported'.format(name))
            

[docs]    def build_features(self):
        """
        Compute features
        
        :return: computed features
        :rtype: Pandas.DataFrame
        """
        assert self.__g is not None, 'Call self.build_graph before building features'
        
        logger.info('Start building features {}'.format(list(self.__features)))
        logger.info('Using config:\n {}'.format(self.__config))
        res = {}
        for name in self.__features:
            res[name] = self.__build_feature(name)
        # map back to node_id_int (raw)
        res['node_id_int'] = list(self.int2raw(nd) for nd in self.__g.iterNodes())
        df = pd.DataFrame(res)
        logger.info('Finished building features')
        
        return df
