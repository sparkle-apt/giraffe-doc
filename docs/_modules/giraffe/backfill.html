[docs]class BiBackfiller:
    """
    This is a class for backfilling graph-based features on bipartite graphs. The process of backfilling 
    is conducted in an on-the-fly fashion. Given a dataframe of entities to query graph features, 
    ``df_query``, the backfiller iteratively builds graphs, computes features, and appends calculated features 
    to ``df_query``. More specifically, the backfiller first calls graph data builder to construct a graph 
    based on data with time window ``[start_date, start_date + window_size[``. Then, the backfiller calls 
    graph feature builder to compute features on the built graph. Once this is done, features are joined to 
    dataframe of query entities subject to the condition that the query entities must be within time window  
    ``[start_date + window_size + gap_size - 1, start_date + window_size + step_size + gap_size - 1[``, 
    where ``gap_size`` indicates the number of days of delay for data. The backfiller then slides the window 
    and construct a second graph based on data with time window 
    ``[start_date + step_size, start_date + window_size + step_size[``. The remaining process repeats as for 
    the first round of iteration.
    """
    
    def __init__(
        self, start_date, end_date, window_size, step_size, 
        df_query, time_col, entity_cols, gd_builder, gf_builder, 
        base_path, gap_size=1, entity_id_map=None, batch_size=10):
        """
        :param str start_date: backfill start date (inclusive)
        :param str end_date: backfill end date (exclusive)
        :param int window_size: size of time window in days
        :param int step_size: step size in days
        :param DataFrame df_query: contains columns of ``entity_cols`` as well ``time_col``
        :param str time_col: name of time col in ``df_query`` for time selection
        :param list entity_cols: col names in ``df_query`` of query entity
        :param gd_builder: :py:class:`BiGraphDataBuilder`
        :param gf_builder: :py:class:`GraphFeatureBuilder`
        :param str base_path: base path for outputs
        :param int gap_size: gap size in days, defaults to 1
        :param DataFrame entity_id_map: produced by calling 
            :py:func:`giraffe.graph_data_builder.BiGraphDataBuilder.get_entity_id_map_data`, defaults to ``None``
        :param int batch_size: batch size for saving checkpoints, defaults to 10
        """
        
        self.start_date = start_date
        self.end_date = end_date
        self.window_size = window_size
        self.step_size = step_size
        self.gap_size = gap_size
        self.entity_cols = entity_cols
        self.time_col = time_col
        self.df_query = df_query
        self.gd_builder = gd_builder
        self.gf_builder = gf_builder
        self.base_path = base_path
        
        self.__batch_size = batch_size
        self.__output_dir = os.path.join(base_path, 'data')
        self.graph_raw_data_path = os.path.join(
            self.base_path, 
            'data', 
            'edge_list_tmp.txt'
        )
        
        self.entity_id_map = gd_builder.get_entity_id_map_data() if entity_id_map is None else entity_id_map
        
    def __shift_date(self, dt, fmt='%Y-%m-%d', ftime=True, **kwargs):
        """
        :param bool ftime: ``True`` to call ``strftime`` to format date object
        :param dict kwargs: kwargs passed to datetime.timedelta; for instance, ``days=1``

        :return: shifted date
        """

        dt_obj = datetime.datetime.strptime(dt, fmt)
        shifted_dt = dt_obj + datetime.timedelta(**kwargs)
        if ftime:
            res = shifted_dt.strftime(fmt)
        else:
            res = shifted_dt

        return res
    
    def __get_graph_data(self, start_date, end_date, **kwargs):    
        _ = self.gd_builder.get_graph_data(
            output_path=self.graph_raw_data_path,
            start_date=start_date, 
            end_date=end_date,
            **kwargs
        )

        return
        

[docs]    def run(self):
        """
        Run to start backfilling
        
        :return: backfilled graph features
        :rtype: Pandas.DataFrame
        """
        prev_start_date_w = None
        prev_end_date_w = None
        cur_start_date_w = self.start_date
        cur_end_date_w = self.__shift_date(self.start_date, days=self.window_size)
        df_list = []
        
        i = 1
        while True:
            if cur_end_date_w >= self.end_date:
                break
            
            logger.info('Iteration {}'.format(i))
            

            lb = self.__shift_date(cur_end_date_w, days=self.gap_size - 1) 
            ub = self.__shift_date(cur_end_date_w, days=self.step_size + self.gap_size - 1) # exclusive

            # init
            if prev_start_date_w is None:
                self.__get_graph_data(cur_start_date_w, cur_end_date_w, force=True)
                self.gf_builder.build_graph(self.graph_raw_data_path)
            else:            
                df_edge_list_delta = self.gd_builder.get_graph_delta_data(
                    ref_start_date=prev_start_date_w, 
                    ref_end_date=prev_end_date_w, 
                    start_date=cur_start_date_w, 
                    end_date=cur_end_date_w
                )
                self.gf_builder.update_graph(df_edge_list_delta)
                logger.info(nk.overview(self.gf_builder.g))

            # features
            df_features = self.gf_builder.build_features()
            # mapping
            df_features = df_features.merge(
                self.entity_id_map, left_on='node_id_int', right_on='node_id_int', how='inner'
            ).drop('node_id_int', axis=1)

            # join to df
            logger.info('lower bound: {}, upper bound: {}'.format(lb, ub))
            df_slice = self.df_query.loc[(self.df_query[self.time_col] >= lb) & (self.df_query[self.time_col] < ub)]

            entity_graph_prefix = '_'.join(self.entity_cols)
            for join_key in self.entity_cols:
                df_features_slice = df_features.loc[df_features['node_typ'] == join_key]\
                    .rename(columns={'node_id': join_key}).drop('node_typ', axis=1)
                _cols = df_features_slice.columns
                df_features_slice = df_features_slice.rename(
                    columns={
                        col:'{}_{}_{}'.format(entity_graph_prefix, join_key, col) for col in _cols if col != join_key
                    }
                )
                df_slice = df_slice.merge(
                    df_features_slice, on=join_key, how='left'
                )

            df_list.append(df_slice)

            if i % self.__batch_size == 1:
                df_res = pd.concat(df_list)
                output_path = os.path.join(self.__output_dir, 'features_1_{}.csv'.format(i))
                df_res.to_csv(output_path, index=False, header=True)

            # update
            i += 1
            prev_start_date_w = cur_start_date_w
            prev_end_date_w = cur_end_date_w
            cur_start_date_w = self.__shift_date(prev_start_date_w, days=self.step_size)
            cur_end_date_w = self.__shift_date(prev_end_date_w, days=self.step_size)
        
        df_res = pd.concat(df_list)
        
        return df_res
